import (

    // Process event
    result, err := fh.processEvent(event)

        return nil, err

    return result, nil
}

                Source:    "user-service",
                Type:      "UserCreated",
                Timestamp: time.Now(),
            },
        },

    fmt.Printf("Serverless Function: %+v\n", function)
    fmt.Printf("Event-Driven Architecture: %+v\n", eda)
